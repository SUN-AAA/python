#if-elif-else문
if num1 == num2:
	print(f"{num1} = {num2}")
elif num1 > num2:
	print(f"{num1} > {num2}")
else
	print(f"{num1} < {num2}")

문자열 함수
#다양한 문자열 조작 함수
my_string = "hello world with ,my dear friends"
#my_string.find()와 같이 문자열 뒤에 .을 붙이고 함수를 사용한다.

#1. startswith() : 문자열이 괄호 안 문자열로 시작되는지 확인하고 True, False 값 반환
print(my_string.startswith('a'))
print(my_string.startswith('h'))

#2. endswith() : 문자열이 괄호 안 문자열로 끝나는지 확인하고 True, False 값 반환
print(my_string.endswith('s'))
print(my_string.endswith('a'))

#3. [인덱스1:인덱스2] : 인덱스1 위치의 문자부터 인덱스2 위치의 문자로 새로운 문자열을 만듦 
sub_word = my_string[2:5]
print(sub_word)

sub_word = my_string[:5]
print(sub_word)

sub_word = my_string[:]
print(sub_word)

#4. strip() : 괄호 안의 문자열을 제거. 괄호 안이 공백이면 문자열 양 끝의 공백을 제거함
print(my_string.strip())
print(my_string.strip('h'))
print(my_string.strip("hello"))

print(my_string.lstrip('h'))
print(my_string.rstrip('s'))

#5. replace(문자열1, 문자열2) : 문자열1을 문자열2로 치환함 
new_string = "Hot coffee"
print(new_string)
print(new_string.replace("Hot", "Cold"))

#6. index(문자열) : 괄호 안 문자열의 위치(인덱스)를 알려줌, 해당 문자열이 없으면 error 출력
print(my_string.index("hello"))

#7. find(문자열) : 주어진 문자열을 찾아서 인덱스 반환, 없으면 -1 반환
 print(my_string.find("hello"))
 print(my_string.rfind("friends"))

#8. upper() : 문자열을 대문자로 변환된 새로운 문자열 반환
print(my_string.upper())

#9. lower() : 문자열을 소문자로 변환된 새로운 문자열 반환
print(my_string.lower())

#10. swapcase() : 문자열의 대문자는 소문자로 변환, 소문자는 대문자로 변환된 새로운 문자열 반환
print(my_string.swapcase())

#11. count(str) : 문자열에서 주어진 문자열인 str이 몇 번 나오는지 반환
string = "hello world"
print(string.count("l"))

#기타
#count(str, start, end) : 문자열에서 주어진 문자열인 str이 몇 번 나오는지 찾는데 start 부터 end – 1 인덱스내에서만 검색

#isalnum() : 문자열이 알파벳과 숫자로만 구성되어 있는지 확인
engstr = "Hi friends"
numstr = "222"

print(numstr.isalnum())
print(engstr.isalnum())

#isalpha() : 문자열이 알파벳만으로 구성되어 있는지 확인
print(engstr.isalpha())
print(numstr.isalnum())
